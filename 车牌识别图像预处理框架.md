# 车牌识别图像预处理框架

## 1. 引言

车牌识别系统在智能交通、安防监控等领域具有广泛应用。然而，实际场景中的光照变化、车牌倾斜、噪声干扰等因素严重影响了识别准确率。本文档详细介绍了一套完整的车牌图像预处理框架，旨在提升LPRNet等车牌识别模型的识别性能。

## 2. CCPD数据集分析

### 2.1 数据集结构

CCPD（Chinese City Parking Dataset）是一个大规模的中国车牌数据集，包含CCPD2019和CCPD2020两个版本。数据集结构如下：

- **CCPD2019**：包含多种场景的车牌图像，如正常、倾斜、模糊等
- **CCPD2020**：增加了新能源车牌数据

### 2.2 文件名编码规则

CCPD数据集的文件名包含丰富的标注信息，格式如下：

```
[区域][角度][亮度][模糊度][遮挡度][车牌类型][颜色][边框坐标][四个角点坐标][车牌号].jpg
```

例如：`02-95_113-154&383_386&473-386&473_177&454_154&383_363&402-0_0_22_27_27_33_16-37-15.jpg`

### 2.3 坐标信息分析

文件名中的坐标信息包括：

- **边框坐标**：`x1&y1_x2&y2`，表示车牌的边界框
- **四个角点坐标**：`x1&y1_x2&y2_x3&y3_x4&y4`，按顺时针顺序排列

### 2.4 原始裁剪的问题

直接使用文件名中的坐标进行裁剪存在以下问题：

1. **倾斜问题**：车牌可能存在不同程度的倾斜
2. **光照问题**：不同场景下光照条件差异大
3. **噪声问题**：图像中可能存在各种噪声干扰
4. **边缘问题**：裁剪边界可能包含多余信息

### 3.2 倾斜校正

#### 3.2.1 基于边缘检测的霍夫变换

```python
def correct_skew(plate_image):
    """基于边缘检测和霍夫变换校正车牌倾斜"""
    # 转换为灰度图
    gray = cv2.cvtColor(plate_image, cv2.COLOR_BGR2GRAY)
    
    # 边缘检测
    edges = cv2.Canny(gray, 50, 150, apertureSize=3)
    
    # 霍夫变换检测直线
    lines = cv2.HoughLines(edges, 1, np.pi/180, 100)
    
    if lines is not None:
        # 计算平均角度
        angles = []
        for line in lines:
            rho, theta = line[0]
            angle = theta * 180 / np.pi
            if angle < 90:
                angle += 90
            else:
                angle -= 90
            angles.append(angle)
        
        if angles:
            avg_angle = np.mean(angles)
            
            # 获取图像中心点
            (h, w) = plate_image.shape[:2]
            center = (w // 2, h // 2)
            
            # 计算旋转矩阵
            M = cv2.getRotationMatrix2D(center, avg_angle, 1.0)
            
            # 执行旋转
            corrected = cv2.warpAffine(plate_image, M, (w, h), 
                                     flags=cv2.INTER_CUBIC, 
                                     borderMode=cv2.BORDER_REPLICATE)
            return corrected
    
    return plate_image
```

#### 3.2.2 基于最小面积矩形的校正

```python
def correct_skew_min_area(plate_image):
    """基于最小面积矩形校正车牌倾斜"""
    # 转换为灰度图
    gray = cv2.cvtColor(plate_image, cv2.COLOR_BGR2GRAY)
    
    # 二值化
    _, thresh = cv2.threshold(gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    
    # 查找轮廓
    contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    
    if contours:
        # 找到最大轮廓
        largest_contour = max(contours, key=cv2.contourArea)
        
        # 计算最小面积矩形
        rect = cv2.minAreaRect(largest_contour)
        box = cv2.boxPoints(rect)
        box = np.int0(box)
        
        # 计算旋转角度
        angle = rect[2]
        if angle < -45:
            angle += 90
        
        # 获取图像中心点
        (h, w) = plate_image.shape[:2]
        center = (w // 2, h // 2)
        
        # 计算旋转矩阵
        M = cv2.getRotationMatrix2D(center, angle, 1.0)
        
        # 执行旋转
        corrected = cv2.warpAffine(plate_image, M, (w, h), 
                                 flags=cv2.INTER_CUBIC, 
                                 borderMode=cv2.BORDER_REPLICATE)
        return corrected
    
    return plate_image
```

### 3.3 光照预处理

#### 3.3.1 直方图均衡化

```python
def histogram_equalization(plate_image):
    """应用直方图均衡化改善光照"""
    # 转换为YUV颜色空间
    yuv = cv2.cvtColor(plate_image, cv2.COLOR_BGR2YUV)
    
    # 对Y通道进行直方图均衡化
    yuv[:, :, 0] = cv2.equalizeHist(yuv[:, :, 0])
    
    # 转换回BGR颜色空间
    return cv2.cvtColor(yuv, cv2.COLOR_YUV2BGR)
```

#### 3.3.2 自适应直方图均衡化（CLAHE）

```python
def clahe_improvement(plate_image):
    """应用CLAHE改善光照"""
    # 转换为YUV颜色空间
    yuv = cv2.cvtColor(plate_image, cv2.COLOR_BGR2YUV)
    
    # 创建CLAHE对象
    clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
    
    # 对Y通道应用CLAHE
    yuv[:, :, 0] = clahe.apply(yuv[:, :, 0])
    
    # 转换回BGR颜色空间
    return cv2.cvtColor(yuv, cv2.COLOR_YUV2BGR)
```

#### 3.3.3 伽马校正

```python
def gamma_correction(plate_image, gamma=1.0):
    """应用伽马校正调整图像亮度"""
    # 构建伽马 lookup 表
    invGamma = 1.0 / gamma
    table = np.array([((i / 255.0) ** invGamma) * 255 for i in np.arange(0, 256)]).astype(np.uint8)
    
    # 应用伽马校正
    return cv2.LUT(plate_image, table)
```

#### 3.3.4 Retinex算法

```python
def retinex_enhance(plate_image, sigma_list=[15, 80, 250]):
    """应用Retinex算法增强图像"""
    # 转换为灰度图
    gray = cv2.cvtColor(plate_image, cv2.COLOR_BGR2GRAY)
    
    # 对数变换
    gray = np.float64(gray) + 1.0
    log_gray = np.log(gray)
    
    # 多尺度高斯模糊
    retinex = np.zeros_like(log_gray)
    for sigma in sigma_list:
        blur = cv2.GaussianBlur(gray, (0, 0), sigma)
        log_blur = np.log(np.float64(blur) + 1.0)
        retinex += log_gray - log_blur
    
    # 平均
    retinex = retinex / len(sigma_list)
    
    # 指数变换
    retinex = np.exp(retinex) - 1.0
    
    # 归一化
    retinex = cv2.normalize(retinex, None, 0, 255, cv2.NORM_MINMAX)
    retinex = np.uint8(retinex)
    
    # 转换回BGR
    return cv2.cvtColor(retinex, cv2.COLOR_GRAY2BGR)
```

### 3.4 图像增强

#### 3.4.1 二值化

```python
def binarize_plate(plate_image):
    """对车牌图像进行二值化处理"""
    # 转换为灰度图
    gray = cv2.cvtColor(plate_image, cv2.COLOR_BGR2GRAY)
    
    # 自适应二值化
    binary = cv2.adaptiveThreshold(gray, 255, 
                                 cv2.ADAPTIVE_THRESH_GAUSSIAN_C, 
                                 cv2.THRESH_BINARY_INV, 11, 2)
    
    # 转换回BGR
    return cv2.cvtColor(binary, cv2.COLOR_GRAY2BGR)
```

#### 3.4.2 形态学操作

```python
def morphology_operations(plate_image):
    """应用形态学操作增强车牌图像"""
    # 转换为灰度图
    gray = cv2.cvtColor(plate_image, cv2.COLOR_BGR2GRAY)
    
    # 二值化
    _, binary = cv2.threshold(gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    
    # 定义结构元素
    kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (3, 3))
    
    # 膨胀操作
    dilated = cv2.dilate(binary, kernel, iterations=1)
    
    # 腐蚀操作
    eroded = cv2.erode(dilated, kernel, iterations=1)
    
    # 转换回BGR
    return cv2.cvtColor(eroded, cv2.COLOR_GRAY2BGR)
```

def gaussian_blur(plate_image, ksize=(3, 3), sigmaX=0):

```python
def sharpen_plate(plate_image):
    """锐化车牌图像增强边缘"""
    # 定义锐化核
    kernel = np.array([[-1, -1, -1],
                       [-1, 9, -1],
                       [-1, -1, -1]])
    
    # 应用卷积
    return cv2.filter2D(plate_image, -1, kernel)
```

## 4. 完整预处理流程

### 4.1 集成预处理函数

```python
def preprocess_plate(plate_image):
    """完整的车牌图像预处理流程"""
    # 1. 倾斜校正
    plate_image = correct_skew_min_area(plate_image)
    
    # 2. 光照预处理
    plate_image = clahe_improvement(plate_image)
    
    # 3. 图像增强
    plate_image = gaussian_blur(plate_image)
    plate_image = sharpen_plate(plate_image)
    
    # 4. 尺寸调整
    plate_image = resize_plate(plate_image)
    
    return plate_image
```
